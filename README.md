### Highlighted Value Method
```js
highlightedVal: function (value) {
    if (value) {
        var regEx = new RegExp(this.query, 'i'),
            foundValue = value.match(regEx);
        return value.replace(foundValue, '<strong class="highlight">' + foundValue + '</strong>');
    } else {
        return value;
    }
}
```
